93  /* size_t invoke_hash(const TYPE & i_object) - Computes the hash of an object.
94  - If a the call hash_func(i_object) is legal, it is used to compute the hash. This function
95  should be defined in the namespace that contains TYPE (it will use ADL). If such function exits,
96  its return type must be size_t.
97  - Otherwise std::hash<TYPE> is used to compute the hash
98  see http://stackoverflow.com/questions/257288/is-it-possible-to-write-a-c-template-to-check-for-a-functions-existence */
99  template <typename> struct sfinae_true : std::true_type { };
100  template <typename TYPE> static sfinae_true<decltype(hash_func(std::declval<TYPE>()))> has_hash_func_impl(int);
101  template <typename TYPE> static std::false_type has_hash_func_impl(long);
102  template <typename TYPE> inline size_t invoke_hash_func_impl(const TYPE & i_object, std::true_type)
103  {
104  static_assert(std::is_same< decltype(hash_func(i_object)), size_t >::value,
105  "if the hash_func() exits for this type, then it must return a size_t");
106  return hash_func(i_object);
107  }
108  template <typename TYPE> inline size_t invoke_hash_func_impl(const TYPE & i_object, std::false_type)
109  {
110  return std::hash<TYPE>()(i_object);
111  }
112  template <typename TYPE> inline size_t invoke_hash(const TYPE & i_object)
113  {
114  return invoke_hash_func_impl(i_object, decltype(has_hash_func_impl<TYPE>(0))());
115  }

117  /* DERIVED * down_cast<DERIVED*>(BASE *i_base_ptr) - down cast from a base class to a derived, assuming
118  that the cast is legal. A static_cast is used if it is possible. Otherwise, if a virtual base is
119  involved, dynamic_cast is used. */
120  template <typename DERIVED, typename BASE> static sfinae_true<decltype(
121  static_cast<DERIVED>(std::declval<BASE>()))> can_static_cast_impl(int);
122  template <typename DERIVED, typename BASE> static std::false_type can_static_cast_impl(long);
123  template <typename DERIVED, typename BASE>
124  inline DERIVED down_cast_impl(BASE i_base_ptr, std::true_type)
125  {
126  return static_cast<DERIVED>(i_base_ptr);
127  }
128  template <typename DERIVED, typename BASE>
129  inline DERIVED down_cast_impl(BASE i_base_ptr, std::false_type)
130  {
131  return dynamic_cast<DERIVED>(i_base_ptr);
132  }
133  template <typename DERIVED, typename BASE>
134  inline DERIVED down_cast(BASE i_base_ptr)
135  {
136  static_assert(std::is_pointer<DERIVED>::value, "DERIVED must be a pointer");
137  static_assert(std::is_pointer<BASE>::value, "BASE must be a pointer");
138 
139  using BaseNaked = typename std::decay<typename std::remove_pointer<BASE>::type>::type;
140  using DerivedNaked = typename std::decay<typename std::remove_pointer<DERIVED>::type>::type;
141  static_assert(std::is_same< BaseNaked, void >::value || std::is_same< BaseNaked, DerivedNaked >::value ||
142  std::is_base_of< BaseNaked, DerivedNaked >::value, "*BASE must be void, the same or a base a base of *DERIVED");
143 
144  return down_cast_impl<DERIVED>(i_base_ptr, decltype(can_static_cast_impl<DERIVED, BASE>(0))());

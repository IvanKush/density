92  /* size_t invoke_hash(const TYPE & i_object) - Computes the hash of an object.
93  - If a the call hash_func(i_object) is legal, it is used to compute the hash. This function
94  should be defined in the namespace that contains TYPE (it will use ADL). If such function exits,
95  its return type must be size_t.
96  - Otherwise std::hash<TYPE> is used to compute the hash
97  see http://stackoverflow.com/questions/257288/is-it-possible-to-write-a-c-template-to-check-for-a-functions-existence */
98  template <typename> struct sfinae_true : std::true_type
99  {
100  };
101  template <typename TYPE>
102  static sfinae_true<decltype(hash_func(std::declval<TYPE>()))> has_hash_func_impl(int);
103  template <typename TYPE> static std::false_type has_hash_func_impl(long);
104  template <typename TYPE>
105  inline size_t invoke_hash_func_impl(const TYPE & i_object, std::true_type)
106  {
107  static_assert(
108  std::is_same<decltype(hash_func(i_object)), size_t>::value,
109  "if the hash_func() exits for this type, then it must return a size_t");
110  return hash_func(i_object);
111  }
112  template <typename TYPE>
113  inline size_t invoke_hash_func_impl(const TYPE & i_object, std::false_type)
114  {
115  return std::hash<TYPE>()(i_object);
116  }
117  template <typename TYPE> inline size_t invoke_hash(const TYPE & i_object)
118  {
119  return invoke_hash_func_impl(i_object, decltype(has_hash_func_impl<TYPE>(0))());
120  }

122  /* DERIVED * down_cast<DERIVED*>(BASE *i_base_ptr) - down cast from a base class to a derived, assuming
123  that the cast is legal. A static_cast is used if it is possible. Otherwise, if a virtual base is
124  involved, dynamic_cast is used. */
125  template <typename DERIVED, typename BASE>
126  static sfinae_true<decltype(static_cast<DERIVED>(std::declval<BASE>()))>
127  can_static_cast_impl(int);
128  template <typename DERIVED, typename BASE>
129  static std::false_type can_static_cast_impl(long);
130  template <typename DERIVED, typename BASE>
131  inline DERIVED down_cast_impl(BASE i_base_ptr, std::true_type)
132  {
133  return static_cast<DERIVED>(i_base_ptr);
134  }
135  template <typename DERIVED, typename BASE>
136  inline DERIVED down_cast_impl(BASE i_base_ptr, std::false_type)
137  {
138  return dynamic_cast<DERIVED>(i_base_ptr);
139  }
140  template <typename DERIVED, typename BASE> inline DERIVED down_cast(BASE i_base_ptr)
141  {
142  static_assert(std::is_pointer<DERIVED>::value, "DERIVED must be a pointer");
143  static_assert(std::is_pointer<BASE>::value, "BASE must be a pointer");
144 
145  using BaseNaked = typename std::decay<typename std::remove_pointer<BASE>::type>::type;
146  using DerivedNaked =
147  typename std::decay<typename std::remove_pointer<DERIVED>::type>::type;
148  static_assert(
149  std::is_same<BaseNaked, void>::value ||
150  std::is_same<BaseNaked, DerivedNaked>::value ||
151  std::is_base_of<BaseNaked, DerivedNaked>::value,
152  "*BASE must be void, DERIVED or a base of *DERIVED");
153 
154  return down_cast_impl<DERIVED>(
155  i_base_ptr, decltype(can_static_cast_impl<DERIVED, BASE>(0))());
156  }

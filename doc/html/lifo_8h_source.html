93  void * allocate(size_t i_size)
94  {
95  DENSITY_ASSERT(i_size % alignment == 0);
96 
97  auto const new_top = m_top + i_size;
98  auto const new_offset =
99  new_top - uint_lower_align(m_top, UNDERLYING_ALLOCATOR::page_alignment);
100  if (!DENSITY_LIKELY(new_offset < UNDERLYING_ALLOCATOR::page_size))
101  {
102  // page overflow
103  return allocate_slow_path(i_size);
104  }
105  else
106  {
107  // advance m_top
108  DENSITY_ASSERT_INTERNAL(i_size <= UNDERLYING_ALLOCATOR::page_size);
109  auto const new_block = reinterpret_cast<void *>(m_top);
110  m_top = new_top;
111  return new_block;

171  void * reallocate(void * i_block, size_t i_old_size, size_t i_new_size)
172  {
173  // branch depending on the old block: this check detects page switches and external blocks
174  if (same_page(i_block, reinterpret_cast<void *>(m_top)))
175  {
176  // the following call sets the top only when it is sure to not throw
177  auto const new_block = set_top_and_allocate(i_block, i_new_size);
178 
179  copy(i_block, i_old_size, new_block, i_new_size);
180  return new_block;
181  }
182  else
183  {
184  if (
185  (m_top & (UNDERLYING_ALLOCATOR::page_alignment - 1)) == sizeof(PageHeader) &&
186  same_page(reinterpret_cast<PageHeader *>(m_top)[-1].m_prev_page, i_block))
187  {
188  auto const page_to_deallocate = reinterpret_cast<void *>(m_top);
189  DENSITY_ASSERT_INTERNAL(!same_page(page_to_deallocate, i_block));
190 
191  // the following call sets the top only when it is sure to not throw
192  auto const new_block = set_top_and_allocate(i_block, i_new_size);
193 
194  copy(i_block, i_old_size, new_block, i_new_size);
195 
196  UNDERLYING_ALLOCATOR::deallocate_page(page_to_deallocate);
197  return new_block;
198  }
199  else
200  {
201  // the old block is external
202  auto const new_block = allocate(i_new_size);
203  copy(i_block, i_old_size, new_block, i_new_size);
204  UNDERLYING_ALLOCATOR::deallocate(i_block, i_old_size, alignment);

243  DENSITY_NO_INLINE void * allocate_slow_path(size_t i_size)
244  {
245  DENSITY_ASSERT_INTERNAL(i_size % alignment == 0);
246  if (i_size < UNDERLYING_ALLOCATOR::page_size / 2)
247  {
248  // allocate a new page
249  auto const new_page = UNDERLYING_ALLOCATOR::allocate_page();
250  DENSITY_ASSERT_INTERNAL(new_page != nullptr);
251  auto const new_header = new (new_page) PageHeader;
252  new_header->m_prev_page = reinterpret_cast<void *>(m_top);
253  m_top = reinterpret_cast<uintptr_t>(new_header + 1) + i_size;
254  return new_header + 1;
255  }
256  else
257  {
258  // external block
259  return UNDERLYING_ALLOCATOR::allocate(i_size, alignment);
260  }

286  void * set_top_and_allocate(void * const i_current_top, size_t const i_size)
287  {
288  DENSITY_ASSERT_INTERNAL(i_size % alignment == 0);
289 
290  auto const current_top = reinterpret_cast<uintptr_t>(i_current_top);
291 
292  // we have to procrastinate the assignment of m_top until we have allocated the page or the external block
293  auto const new_top = current_top + i_size;
294  auto const new_offset =
295  new_top - uint_lower_align(current_top, UNDERLYING_ALLOCATOR::page_alignment);
296  if (new_offset < UNDERLYING_ALLOCATOR::page_size)
297  {
298  DENSITY_ASSERT_INTERNAL(i_size <= UNDERLYING_ALLOCATOR::page_size);
299  auto const new_block = i_current_top;
300  m_top = new_top;
301  return new_block;
302  }
303  else if (i_size < UNDERLYING_ALLOCATOR::page_size / 2)
304  {
305  // allocate a new page
306  auto const new_page =
307  static_cast<PageHeader *>(UNDERLYING_ALLOCATOR::allocate_page());
308  new_page->m_prev_page = reinterpret_cast<void *>(current_top);
309  m_top = reinterpret_cast<uintptr_t>(new_page + 1) + i_size;
310  return new_page + 1;
311  }
312  else
313  {
314  // external block
315  auto const new_external_block = UNDERLYING_ALLOCATOR::allocate(i_size, alignment);
316  m_top = reinterpret_cast<uintptr_t>(i_current_top);
317  return new_external_block;
318  }
319  }

322  void * i_old_block, size_t i_old_size, void * i_new_block, size_t i_new_size) noexcept
323  {
324  if (i_old_block != i_new_block)

330  memcpy(i_new_block, i_old_block, size_to_copy);

821  template <typename... CONSTRUCTION_PARAMS>
822  lifo_array(size_t i_size, const CONSTRUCTION_PARAMS &... i_construction_params)
823  : detail::LifoArrayImpl<TYPE>(i_size)
824  {
825  size_t element_index = 0;
826  try
827  {
828  for (; element_index < i_size; element_index++)
829  {
830  new (m_elements + element_index) TYPE(i_construction_params...);
831  }
832  }
833  catch (...)
834  {
835  // destroy all the elements constructed till now, and then deallocate
836  auto it = m_elements + element_index;
837  it--;
838  for (; it >= m_elements; it--)
839  {
840  it->TYPE::~TYPE();
841  }
842  throw;
843  }
844  }

454  inline void * aligned_allocate(size_t i_size, size_t i_alignment, size_t i_alignment_offset = 0)
455  {
456  DENSITY_ASSERT(i_alignment > 0 && is_power_of_2(i_alignment));
457  DENSITY_ASSERT(i_alignment_offset <= i_size);
458 
459  void * user_block;
460  if (i_alignment <= detail::MaxAlignment && i_alignment_offset == 0)
461  {
462  user_block = operator new(i_size);
463  }
464  else
465  {
466  // reserve an additional space in the block equal to the max(i_alignment, sizeof(AlignmentHeader) - sizeof(void*) )
467  size_t const extra_size =
468  detail::size_max(i_alignment, sizeof(detail::AlignmentHeader));
469  size_t const actual_size = i_size + extra_size;
470  auto const complete_block = operator new(actual_size);
471  user_block = address_lower_align(
472  address_add(complete_block, extra_size), i_alignment, i_alignment_offset);
473  detail::AlignmentHeader & header =
474  *(static_cast<detail::AlignmentHeader *>(user_block) - 1);
475  header.m_block = complete_block;
476  DENSITY_ASSERT_INTERNAL(
477  user_block >= complete_block &&
478  address_add(user_block, i_size) <= address_add(complete_block, actual_size));
479  }
480  return user_block;
481  }

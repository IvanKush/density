449  inline void * aligned_allocate(size_t i_size, size_t i_alignment, size_t i_alignment_offset = 0)
450  {
451  DENSITY_ASSERT(i_alignment > 0 && is_power_of_2(i_alignment));
452  DENSITY_ASSERT(i_alignment_offset <= i_size);
453 
454  void * user_block;
455  if (i_alignment <= detail::MaxAlignment && i_alignment_offset == 0)
456  {
457  user_block = operator new (i_size);
458  }
459  else
460  {
461  // reserve an additional space in the block equal to the max(i_alignment, sizeof(AlignmentHeader) - sizeof(void*) )
462  size_t const extra_size = detail::size_max(i_alignment, sizeof(detail::AlignmentHeader));
463  size_t const actual_size = i_size + extra_size;
464  auto const complete_block = operator new (actual_size);
465  user_block = address_lower_align(address_add(complete_block, extra_size), i_alignment, i_alignment_offset);
466  detail::AlignmentHeader & header = *(static_cast<detail::AlignmentHeader*>(user_block) - 1);
467  header.m_block = complete_block;
468  DENSITY_ASSERT_INTERNAL(user_block >= complete_block &&
469  address_add(user_block, i_size) <= address_add(complete_block, actual_size));
470  }
471  return user_block;
472  }
